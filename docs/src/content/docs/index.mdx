## Quick Start

Install SpectraDB and run your first query in under 5 minutes:

```rust
use spectradb::{Database, Config};

fn main() -> spectradb::Result<()> {
    let db = Database::open("my_data", Config::default())?;

    // Create a table
    db.sql("CREATE TABLE users (id TEXT, name TEXT, email TEXT)", &[])?;

    // Insert data â€” every write is an immutable fact
    db.sql("INSERT INTO users VALUES ('u1', 'Alice', 'alice@co.com')", &[])?;

    // Query with full SQL
    let results = db.sql("SELECT name, email FROM users WHERE id = 'u1'", &[])?;
    println!("{:?}", results);

    Ok(())
}
```
